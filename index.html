// Кэш для селекта let cachedSelectHTML = ""; function buildSelectHTML() { if (cachedSelectHTML) return cachedSelectHTML; let html = '"; cachedSelectHTML = html; return html; } function toggleInput(id, isChecked) { const el = document.getElementById(id); if(isChecked) { el.classList.remove('hidden'); } else { el.classList.add('hidden'); } } // Искры function sparksAt(x, y, count = 10) { for (let i = 0; i < count; i++) { const s = document.createElement("div"); s.className = "spark"; const dx = Math.random() * 120 - 60 + "px"; const dy = Math.random() * 120 - 80 + "px"; s.style.left = x + "px"; s.style.top = y + "px"; s.style.setProperty("--dx", dx); s.style.setProperty("--dy", dy); document.body.appendChild(s); setTimeout(() => s.remove(), 600); } } // Итоги function calcTotals() { let total = 0; document.querySelectorAll(".metal-weight").forEach((inp) => { const v = parseFloat(inp.value); if (!isNaN(v) && v > 0) total += v; }); const trucks = total > 0 ? Math.ceil(total / 15.0) : 0; // Убираем нули: parseFloat(num.toFixed(3)) делает 5.000 -> 5 const prettyTotal = parseFloat(total.toFixed(3)); document.getElementById("sum-weight").innerText = prettyTotal + " ТОНН"; document.getElementById("sum-trucks").innerText = trucks + (trucks === 1 ? " РЕЙС" : " РЕЙСОВ"); updateMainButtonState(); } // Счётчик позиций function updateCount() { const count = document.getElementsByClassName("item-card").length; document.getElementById("count-badge").innerText = count + " ПОЗ."; updateMainButtonState(); } // Управление состоянием главной кнопки function updateMainButtonState() { const btn = document.getElementById("main-btn"); const objName = document.getElementById("objectName").value.trim(); const prodTime = parseInt(document.getElementById("production_time").value, 10); // есть ли хоть одна валидная позиция let hasValidItem = false; document.querySelectorAll(".item-card").forEach((card) => { const code = card.querySelector(".metal-type").value; const weight = parseFloat(card.querySelector(".metal-weight").value); if (code && !isNaN(weight) && weight > 0) hasValidItem = true; }); const ok = objName.length > 0 && Number.isFinite(prodTime) && prodTime > 0 && hasValidItem; btn.disabled = !ok; btn.innerText = ok ? "СФОРМИРОВАТЬ КП" : (objName.length === 0 || !prodTime ? "ЗАПОЛНИТЕ ДАННЫЕ" : "ДОБАВЬТЕ МЕТАЛЛ"); } // Добавить позицию function addItem() { if (tg.HapticFeedback) tg.HapticFeedback.impactOccurred("light"); const container = document.getElementById("items-container"); const div = document.createElement("div"); div.className = "item-card"; div.innerHTML = ` ${buildSelectHTML()}
×
`; container.appendChild(div); updateCount(); // небольшой вау-момент: искры из центра кнопки добавления const addBtn = document.getElementById("add-btn"); const r = addBtn.getBoundingClientRect(); sparksAt(r.left + r.width / 2, r.top + r.height / 2, 12); // автофокус: сначала на select const sel = div.querySelector(".metal-type"); sel.focus(); // когда выбрали профиль — прыгнуть на вес sel.addEventListener("change", () => { const w = div.querySelector(".metal-weight"); w.focus(); }); div.scrollIntoView({ behavior: "smooth", block: "center" }); } // Удалить позицию function removeItem(btn) { if (tg.HapticFeedback) tg.HapticFeedback.impactOccurred("medium"); const card = btn.closest(".item-card"); card.classList.add("removing"); setTimeout(() => { card.remove(); updateCount(); calcTotals(); }, 300); } // Отправка данных function sendData() { const objName = document.getElementById("objectName").value.trim(); const objAddr = document.getElementById("objectAddress").value.trim(); const prodTimeRaw = document.getElementById("production_time").value; const prodTime = parseInt(prodTimeRaw, 10); // Валидация объекта/срока if (!objName || !Number.isFinite(prodTime) || prodTime <= 0) { if (tg.HapticFeedback) tg.HapticFeedback.notificationOccurred("error"); tg.showPopup({ title: "Ошибка", message: 'Укажите "Название объекта" и корректный "Срок изготовления"!' }); return; } let items = []; let isValidItems = true; document.querySelectorAll(".item-card").forEach((card) => { const sel = card.querySelector(".metal-type"); const code = sel.value; const weightInput = card.querySelector(".metal-weight"); const weight = parseFloat(weightInput.value); const nameText = sel.selectedIndex > 0 ? sel.options[sel.selectedIndex].text : ""; if (!code || isNaN(weight) || weight <= 0) { card.style.borderLeftColor = "red"; isValidItems = false; } else { card.style.borderLeftColor = "#FFD700"; // фиксируем максимум 3 знака const fixedWeight = Math.round(weight * 1000) / 1000; items.push({ name: nameText, code: code, weight: fixedWeight }); } }); if (!isValidItems || items.length === 0) { if (tg.HapticFeedback) tg.HapticFeedback.notificationOccurred("warning"); tg.showPopup({ title: "Внимание", message: "Добавьте хотя бы одну позицию металла с весом." }); return; } const isDelivery = document.getElementById("include_delivery").checked; const data = { object_name: objName, object_address: objAddr, items: items, production_time: String(prodTime), // Если доставка выключена, отправляем 0 или специальный флаг (но пока шлем цену) delivery_price: isDelivery ? (document.getElementById("delivery_price").value || 40000) : 0, painting_type: document.getElementById("painting_type").value, include_kmd: document.getElementById("include_kmd").checked, kmd_price: document.getElementById("kmd_price").value || 4500, include_montage: document.getElementById("include_montage").checked, montage_price: document.getElementById("montage_price").value || 47000 }; // вау: искры на кнопке const mainBtn = document.getElementById("main-btn"); const r = mainBtn.getBoundingClientRect(); sparksAt(r.left + r.width / 2, r.top + 10, 16); if (tg.HapticFeedback) tg.HapticFeedback.notificationOccurred("success"); tg.sendData(JSON.stringify(data)); } // Слушатели: пересчёт итогов и состояния кнопки document.addEventListener("input", (e) => { if (e.target && e.target.classList && e.target.classList.contains("metal-weight")) { calcTotals(); } if (e.target && (e.target.id === "objectName" || e.target.id === "production_time")) { updateMainButtonState(); } }); // Кнопка "добавить" document.getElementById("add-btn").addEventListener("click", addItem); // Старт: сразу одна позиция (приятнее, чем пустота) addItem(); calcTotals(); updateMainButtonState();
